        }
        if(this.splashDmg != null && this.splashDmg != null && this.splashDmg != 0) {
            sb.append(" Splash:");
            if (this.splashDmg < 0) {
                sb.append("Heal");
                sb.append(-1 * this.splashDmg);
            } else {
                sb.append(this.splashElement.getValue());
                sb.append(this.splashDmg);
            }
        }
        return sb.toString();
    }
